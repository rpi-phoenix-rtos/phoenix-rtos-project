# Phoenix-RTOS Raspberry Pi 4 early bring-up
#
# This is only the first firmware-facing staging config.
# DTB staging and firmware-handoff support are still incomplete.

arm_64bit=1
armstub=phoenix-armstub8-rpi4.bin
kernel=kernel8.img
initramfs loader.disk 0x08000000
enable_uart=1
uart_2ndstage=1
init_uart_baud=115200
init_uart_clock=48000000
dtoverlay=miniuart-bt
# V3D GPU bring-up (GLQuake roadmap): fake-KMS V3D overlay tells the firmware to
# initialize/ungate the V3D core (its MMIO reads 0xdeadbeef without it) while
# keeping the firmware framebuffer for the HDMI console (full vc4-kms-v3d would
# hand the display to a kernel DRM driver we don't have, breaking fbcon).
dtoverlay=vc4-fkms-v3d
force_turbo=1
core_freq=250
# Skip the firmware bootloader delay (default 1 s) and rainbow splash —
# we are netbooting / development-running, no need for user interaction
# windows.
boot_delay=0
boot_delay_ms=0
disable_poe_fan=1
# Lock GPU reserve at the firmware-minimum that still supports HDMI + mailbox + framebuffer.
# This frees as much DRAM for ARM as the Pi 4 layout allows. Note: `total_mem` cannot
# enlarge memory — it only caps it, so we omit it. Default behaviour is "use all DRAM".
# Pi 4 4GB layout (with gpu_mem=76):
#   0x00000000 - 0x3b3fffff   948 MB   ARM DRAM (chunk 1)
#   0x3b400000 - 0x3fffffff    76 MB   GPU heap + framebuffer (VC4 owns, ARM stays Device)
#   0x40000000 - 0xfbffffff  3008 MB   ARM DRAM (chunk 2) — until now unmapped on ARM
#   0xfc000000 - 0xffffffff    64 MB   BCM2711 peripherals (low-peri mode, MMIO)
gpu_mem=76
hdmi_force_hotplug=1
disable_overscan=1
disable_splash=1
# Force 1920x1080@60 output + framebuffer for the GLQuake 1080p performance target
# (~40fps@1080p, Linux parity). hdmi_group=1 (CEA) mode 16 = 1080p60; the grabber is
# 4K-capable so 1080p is supported. framebuffer_width/height pin the fb to 1920x1080
# (8.3 MB, fits gpu_mem=76); graphmode then reports 1920x1080 to rpi4-fb + Quake.
hdmi_group=1
hdmi_mode=16
framebuffer_width=1920
framebuffer_height=1080
